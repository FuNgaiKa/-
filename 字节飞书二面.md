算法题：

射10箭，0-10环，恰好90环的种类数。

思路：排列组合、递归dfs剪枝

```c++
#include <iostream>
using namespace std;
int ans;
void dfs(int huan,int num){
    if((huan==0&&num<=11)){
        ans++;
        return;
    }
    if(huan<0||num>10){
        return;
    }



    //
    for(int i=0;i<=10;i++){
        huan-=i;
        dfs(huan,num+1);
        huan+=i;
    }

    return;
}
int main() {
    int huan=10;
    ans=0;
    dfs(huan,1);
    cout<<ans<<endl;
}
```
